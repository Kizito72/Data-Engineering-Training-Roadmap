                            // ═══════════════════════════════════════════════════════════ // DATA ENGINEERING ROADMAP — Google Calendar Auto-Setup // Paste this in Google Apps Script (script.google.com) // Run: setupStudySessions() → authorize → done! // ═══════════════════════════════════════════════════════════ // ── CONFIGURATION ────────────────────────────────────────── const CONFIG = { // Study start times (24h format) weekdayStudyHour: 18, // 6:00 PM (after work ends 17:30) weekdayStudyMin: 00, weekendStudyHour: 16, // 4:00 PM (after work ends 16:00) weekendStudyMin: 30, // Session durations (minutes) weekdayDuration: 120, // 2 hours weekendDuration: 240, // 4 hours // Reminder minutes before session reminderPrepMin: 30, // "Get ready" 30 min before reminderStartMin: 0, // Reminder AT session start // Calendar settings calendarName: 'Data Engineering Study', color: CalendarApp.EventColor.CYAN, // Roadmap start (change if needed) startDate: new Date('2025-04-05'), totalWeeks: 12 }; // ── PHASE SCHEDULE ───────────────────────────────────────── const PHASES = [ { weeks: [1,2], label: 'Phase 1 — SQL', emoji: '🗄️' }, { weeks: [3,4], label: 'Phase 1 — Python', emoji: '🐍' }, { weeks: [5,6], label: 'Phase 2 — ETL', emoji: '⚙️' }, { weeks: [7,8], label: 'Phase 2 — Modeling', emoji: '🔄' }, { weeks: [9,10], label: 'Phase 3 — Cloud (GCP)', emoji: '☁️' }, { weeks:[11,12], label: 'Phase 3 — Final Project',emoji: '🚀' } ]; // ── MAIN FUNCTION ─────────────────────────────────────────── function setupStudySessions() { const cal = getOrCreateCalendar(CONFIG.calendarName); // Loop through 12 weeks = 84 days for (let week = 1; week <= CONFIG.totalWeeks; week++) { const phase = getPhase(week); const weekStart = addDays(CONFIG.startDate, (week - 1) * 7); // Monday–Friday (weekdays) for (let d = 0; d < 5; d++) { const day = addDays(weekStart, d); createStudyEvent(cal, day, 'weekday', week, phase); } // Sunday only (Saturday is work day) const sunday = addDays(weekStart, 6); createStudyEvent(cal, sunday, 'weekend', week, phase); } Logger.log('✅ Done! All 12 weeks of study sessions have been added to your Google Calendar with phone reminders.'); } // ── CREATE ONE EVENT ──────────────────────────────────────── function createStudyEvent(cal, date, type, week, phase) { const isWeekend = (type === 'weekend'); const h = isWeekend ? CONFIG.weekendStudyHour : CONFIG.weekdayStudyHour; const m = isWeekend ? CONFIG.weekendStudyMin : CONFIG.weekdayStudyMin; const dur = isWeekend ? CONFIG.weekendDuration : CONFIG.weekdayDuration; const start = new Date(date); start.setHours(h, m, 0, 0); const end = new Date(start); end.setMinutes(end.getMinutes() + dur); const title = `${phase.emoji} ${phase.label} | Week ${week} Study`; const desc = buildDescription(week, isWeekend, phase); const event = cal.createEvent(title, start, end, { description: desc, location: 'Home Study' }); // Set event color by phase event.setColor(CONFIG.color); // ── REMINDERS (phone push notifications) ── event.addPopupReminder(CONFIG.reminderPrepMin); // 30 min: get ready! event.addPopupReminder(CONFIG.reminderStartMin); // 0 min: start now! } // ── HELPER: Get phase info for a given week ───────────────── function getPhase(week) { return PHASES.find(p => p.weeks.includes(week)) || { label: 'Study', emoji: '📚' }; } // ── HELPER: Build event description ──────────────────────── function buildDescription(week, isWeekend, phase) { if (isWeekend) { return `📅 Weekend Study Block — 4 Hours 🎯 Phase: ${phase.label} 📍 Week ${week} of 12 Block 1 (2 hrs): Deep Work / Project Building Block 2 (2 hrs): Revision + Documentation + GitHub Push ⚠️ Rules: Always code. Never just watch tutorials. 🔗 GitHub: Push at least 1 commit today. 📊 Roadmap: https://[your-github-username].github.io/data-roadmap`; } return `📅 Weekday Study Block — 2 Hours 🎯 Phase: ${phase.label} 📍 Week ${week} of 12 Block 1 (60 min): Course Learning Block 2 (60 min): Practice / Project Work ⚠️ Rules: Write 1 example yourself. No copy-paste. 🔗 GitHub: Push even small progress. 📊 Roadmap: https://[your-github-username].github.io/data-roadmap`; } // ── HELPER: Get or create named calendar ──────────────────── function getOrCreateCalendar(name) { const cals = CalendarApp.getCalendarsByName(name); if (cals.length > 0) return cals[0]; return CalendarApp.createCalendar(name, { summary: 'Data Engineering 3-Month Study Plan', color: '#0288d1' }); } // ── HELPER: Add days to date ──────────────────────────────── function addDays(date, n) { const d = new Date(date); d.setDate(d.getDate() + n); return d; } // ── BONUS: Delete all created events (for reset) ──────────── function deleteStudyCalendar() { const cals = CalendarApp.getCalendarsByName(CONFIG.calendarName); if (cals.length > 0) { cals[0].deleteCalendar(); Logger.log('🗑️ Study calendar deleted.'); } }
